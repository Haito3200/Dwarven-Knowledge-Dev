on_adm_development = {
	if = {
		limit = {
			OR = {
				has_terrain = dwarven_hold
				has_terrain = dwarven_hold_surface
			}
			owner = {
				has_country_flag = DA_Disabled_ADK_on_Click
				check_variable = {
					which = ancientDwarvenKnowledge
					value = 1
				}
			}
		}
		add_base_tax = -1
		owner = {
			change_variable = {
				which = ancientDwarvenKnowledge
				value = 1
			}
		}
		add_next_institution_embracement_scaled = -5
	}
}
