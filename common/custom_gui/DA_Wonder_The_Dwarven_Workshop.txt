custom_button = {
	name = DA_The_Dwarven_Workshop_upgrade_button
	tooltip = DA_Wonder_upgrade_button_tt
	potential = {
		always = yes
	}
	trigger = {
		custom_trigger_tooltip = {
			tooltip = DA_Wonder_in_use_tt
			NOT = { has_country_flag = DA_Wonder_The_Dwarven_Workshop_in_use }
		}
		4236 = {
			owned_by = prev
			has_great_project = {
				type = The_Dwarven_Workshop
				tier = 0
			}
		}
		if = {
			limit = {
				4236 = {
					AND = {
						has_great_project = {
							type = The_Dwarven_Workshop
							tier = 0
						}
						NOT = { has_great_project = { type = The_Dwarven_Workshop tier = 1 } }
					}
				}
			}
			4236 = {
				custom_trigger_tooltip = {
					tooltip = DA_tooltip_has_dig_level_4_or_higher

				amount = 10

				amount = 10
			}

				amount = 10
			}

		}
		if = {
			limit = {
				4236 = {
					AND = {

				amount = 40
			}

				amount = 40
			}
			treasury = 7500
		}
		if = {
			limit = {
				4236 = {
					has_great_project = {
						type = The_Dwarven_Workshop
						tier = 3
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = DA_Maxed_wonder_tt
				always = no
			}
		}
	}
	effect = {
		if = {
			limit = {
				4236 = {
					AND = {
						has_great_project = {
							type = The_Dwarven_Workshop
							tier = 0
						}
						NOT = { has_great_project = { type = The_Dwarven_Workshop tier = 1 } }

				}
			}

				amount = 10
			}
			DA_enchanted_building_material_spend = {
				amount = 10
			}

				amount = 10
			}
			add_treasury = -1600

					type = The_Dwarven_Workshop

				country_event = {
					id = DA_Mithradum_event.2
				}
			}
		}
		else_if = {
			limit = {
				4236 = {
					AND = {
						has_great_project = {
							type = The_Dwarven_Workshop
							tier = 1
						}

					}
				}
			}
			DA_alchemy_reagents_spend = {

			}
