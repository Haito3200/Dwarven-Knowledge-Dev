    name = DA_projects_research_Interface_on_button_previous
    tooltip = DA_projects_research_Interface_on_button_previous_tt
    potential = {
        always = yes
    }
    trigger = {
        hidden_trigger = {
            NOT = { has_country_flag = DA_projects_research_Interface_on_arcane_1 }
            NOT = { has_country_flag = DA_projects_research_Interface_on_Steam_1 }
            NOT = { has_country_flag = DA_projects_research_Interface_on_Electrical_1 }

        }
    }
}
