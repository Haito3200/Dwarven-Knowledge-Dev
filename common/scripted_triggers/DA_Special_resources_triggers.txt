DA_alchemy_reagents_total_required = {
	custom_trigger_tooltip = {
		tooltip = DA_alchemy_reagents_total_required_$amount$_tt
		check_variable = {
			which = DA_alchemy_reagents_total
			value = $amount$
		}
	}
}
