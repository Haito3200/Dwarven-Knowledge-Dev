DA_is_Mine_not_hold = {
	custom_trigger_tooltip = {
		tooltip = DA_is_Mine_not_hold_tt
		is_wasteland = no
		OR = {
			has_province_modifier = DA_Mine_1
			has_province_modifier = DA_Mine_2
			has_province_modifier = DA_Mine_3
			has_province_modifier = DA_Mine_4
			has_province_modifier = DA_Mine_5
		}
	}
}
DA_is_Upgradable_mine = {
	OR = {
		AND = {
			base_production = 7
			has_province_modifier = DA_Mine_1
		}
		AND = {
			base_production = 14
			has_province_modifier = DA_Mine_2
		}
		AND = {
			base_production = 21
			has_province_modifier = DA_Mine_3
		}
		AND = {
			base_production = 28
			has_province_modifier = DA_Mine_4
		}
	}
}
