DA_if_flag_give_modifier_effect = {
	if = {
		limit = {
			has_country_flag = DA_Unique_mod.$DA_type$.Flag.$DA_unit$.$DA_modifier$
		}
		add_country_modifier = {
			name = DA_Unique_mod.$DA_type$.$DA_unit$.$DA_modifier$
			duration = -1
		}
	}
}
