
DA_create_Rune_Mage_general = {
	custom_tooltip = Low_Class_Rune_Mage_general_tooltip
	hidden_effect = {
		DA_setup_base_stats_of_generals = yes
		### changes variables according to input
		#input example
		#DA_create_Rune_Mage_general = {
		#	DA_fire = 0
		#	DA_shock = 0
		#	DA_maneuver = 0
		#	DA_siege = 1
		#}
		change_variable = {
			which = DA_base_fire
			value = $DA_fire$
		}
		change_variable = {
			which = DA_base_shock
			value = $DA_shock$
		}
		change_variable = {
			which = DA_base_manuever
			value = $DA_maneuver$
		}
		change_variable = {
			which = DA_base_siege
			value = $DA_siege$
		}
		###

		random_list = {
			50 = { #Male
				set_saved_name = {
					key = root_name_DAgeneral
					type = advisor
					scope = ROOT
				}
				DA_generate_general = {
					DA_trait = $DA_trait$
				}
			}
			50 = { #Female
				set_saved_name = {
					key = root_name_DAgeneral
					type = advisor
					female = yes
					scope = ROOT
				}
				DA_generate_general = {
					DA_trait = $DA_trait$
					female = yes
				}
			}
		}
		###
	}
}
