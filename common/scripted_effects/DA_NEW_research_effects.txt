DA_start_research = {
	custom_tooltip = DA_research_$time$_tt
	set_country_flag = DA_research_in_progress
	set_country_flag = DA_research_flag.$flag$
	hidden_effect = {
		set_variable = {
			which = DA_research_progress
			value = 0
		}
		set_variable = {
			which = DA_research_progress_MAX
			value = $time$
		}
		set_variable = {
			which = DA_research_progress_MAX_temp
			value = $time$
		}
		divide_variable = {
			which = DA_research_progress_MAX_temp
			value = 3
		}
		round_variable = {
			which = DA_research_progress_MAX_temp
			value = 0
		}
		[[arcane]
			subtract_variable = {
				which = DA_research_progress_MAX
				which = DA_research_time_discount_1
			}
		]
		[[steam]
			subtract_variable = {
				which = DA_research_progress_MAX
				which = DA_research_time_discount_2
			}
		]
		[[electric]
			subtract_variable = {
				which = DA_research_progress_MAX
				which = DA_research_time_discount_3
			}
		]
		if = {
			limit = {
				NOT = {
					check_variable = {
						which = DA_research_progress_MAX
						which = DA_research_progress_MAX_temp
					}
				}
			}
			set_variable = {
				which = DA_research_progress_MAX
				which = DA_research_progress_MAX_temp
			}
		}
		country_event = {
			id = DA_General_Research_events.2
			days = 30
		}
	}
	if = {
		limit = {
			[[arcane]
				check_variable = {
					which = DA_research_time_discount_1
					value = 1
				}
			]
			[[steam]
				check_variable = {
					which = DA_research_time_discount_2
					value = 1
				}
			]
			[[electric]
				check_variable = {
					which = DA_research_time_discount_3
					value = 1
				}
			]
		}
		[[arcane]
			custom_tooltip = DA_research_time_discount_1_tt
		]
		[[steam]
			custom_tooltip = DA_research_time_discount_2_tt
		]
		[[electric]
			custom_tooltip = DA_research_time_discount_3_tt
		]
	}
	[[unstoppable]
		custom_tooltip = DA_research_is_unstoppable_tt
		set_country_flag = DA_research_is_unstoppable
	]
}
