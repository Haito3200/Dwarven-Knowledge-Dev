#Window 1 - DA_main_tech_interface
DA_open_first_open_tech_interface = {
	DA_open_main_tech_interface = yes
	DA_open_metallurgy_interface = yes
}

DA_open_main_tech_interface = {
    set_country_flag = DA_show_tech_interface
	if = {
		limit = { DA_has_metallurgy_interface_open = yes }
		hidden_effect = {
			# DA_calculate_metal_provinces = yes
			country_event = { id = DA_Metallurgy_events.0 }
		}
	}
}
DA_close_main_tech_interface = {
    clr_country_flag = DA_show_tech_interface
}
