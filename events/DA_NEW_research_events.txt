country_event = {
	id = DA_General_Research_events.2
	title = DA_General_Research_events.2.t
	desc = DA_General_Research_events.2.d
	picture = INVENTION_eventPicture

	trigger = {
		always = yes
	}

	is_triggered_only = yes

	hidden = yes

	immediate = {
	}

	option = {
		name = DA_General_Research_events.2.a
		export_to_variable = {
			value = modifier:DA_fake_research_speed_gain
			which = DA_research_speed_gain_var
		}
		export_to_variable = {
			value = modifier:DA_fake_research_speed_gain_percentage
			which = DA_research_speed_gain_percentage_var
		}
		multiply_variable = {
			which = DA_research_speed_gain_var
			which = DA_research_speed_gain_percentage_var
		}
		
		multiply_variable = {
			which = DA_research_speed_gain_var
			value = 100
		}
		round_variable = {
			which = DA_research_speed_gain_var
			value = 0
		}
		divide_variable = {
			which = DA_research_speed_gain_var
			value = 100
		}
		change_variable = {
			which = DA_research_progress
			which = DA_research_speed_gain_var
		}
		if = {
			limit = { check_variable = {
				which = DA_research_progress
				which = DA_research_progress_MAX
			} }
			country_event = {
				id = DA_General_Research_events.3
			}
		}
		else = {
			capital_scope = {
				add_province_triggered_modifier = DA_research_increase_workaround
			}
		}

		# this is just for showing the bar
		set_variable = { which = DA_research_progress_percent which = DA_research_progress }
		divide_variable = { which = DA_research_progress_percent which = DA_research_progress_MAX }
		multiply_variable = { DA_research_progress_percent = 100 }
		ai_chance = { factor = 1 }
	}
}
